    <div id="root"></div>

    <script>
        const TYPE_COLOR = {
          Lauf:      { bg: "#e8f5e9", border: "#4caf50", icon: "🏃" },
          Kraft:     { bg: "#e3f2fd", border: "#2196f3", icon: "💪" },
          Mobilität: { bg: "#fff8e1", border: "#ffc107", icon: "🧘" },
        };

        const ALL_WEEKS = [
          { phase:1, week:1, focus:"Gehen mit Laufintervallen",
            sessions:[
              { day:"Mo", type:"Lauf", content:"5 min Gehen → 3× [1 min Laufen / 2 min Gehen] → 5 min Gehen", duration:"~18 min" },
              { day:"Mi", type:"Kraft", duration:"~25 min", blocks:[
                { label:"🦵 Beine & Knie", items:["Wandsitzen 3× 20 Sek","Kniebeugen 3× 10 Wdh (langsam)","Muschel 3× 12 je Seite"] },
                { label:"🧠 Rumpf & Hohlkreuz", items:["Beckenkippung im Liegen 3× 10","Dead Bug 3× 6 je Seite","Plank 3× 20 Sek"] },
                { label:"🔁 Dehnen & Prävention", items:["Hüftbeuger-Stretch kniend 3× 30 Sek","Wadenheben beidbeinig 3× 15 Wdh"] },
              ]},
              { day:"Fr", type:"Lauf", content:"5 min Gehen → 4× [1 min Laufen / 2 min Gehen] → 5 min Gehen", duration:"~20 min" },
            ], tip:"Lauftempo = du kannst dabei noch reden. Wenn nicht → langsamer!" },
          { phase:1, week:2, focus:"Intervalle ausbauen",
            sessions:[
              { day:"Mo", type:"Lauf", content:"5 min Gehen → 4× [1,5 min Laufen / 2 min Gehen] → 5 min Gehen", duration:"~22 min" },
              { day:"Mi", type:"Kraft", duration:"~28 min", blocks:[
                { label:"🦵 Beine & Knie", items:["Wandsitzen 3× 30 Sek","Kniebeugen 3× 12 Wdh","Muschel 3× 15 je Seite"] },
                { label:"🧠 Rumpf & Hohlkreuz", items:["Beckenkippung 3× 10","Dead Bug 3× 8 je Seite","Plank 3× 25 Sek"] },
              ]},
              { day:"Fr", type:"Lauf", content:"5 min Gehen → 5× [1,5 min Laufen / 2 min Gehen] → 5 min Gehen", duration:"~25 min" },
            ], tip:"Herzrasen? Sofort auf Gehen runter." }
          // ... Hier können alle weiteren Wochen analog hinzugefügt werden
        ];

        const VEGAN_NUTRITION_TIPS = [
          { icon:"🏃‍♂️", title:"Grundlagenausdauer & Kohlenhydrate", text:"Kohlenhydrate sind der Haupttreibstoff. Setze auf komplexe, pflanzliche Carbs wie Haferflocken und Quinoa." },
          { icon:"🌱", title:"Vegane High-Quality Proteine", text:"Für die Regeneration benötigst du Tofu, Seitan, Lupinenfilets oder Linsen." }
        ];

        /* MEDIATHEK: Setze hier in den Anführungszeichen bei 'imgUrl' deine echten Bild-Links ein! */
        const EXERCISE_LIBRARY = [
          { title: "1. Wandsitzen (Wall Sit) – Ohne Stuhl", imgUrl: "https://images.unsplash.com/photo-1571019613454-1cb2f99b2d8b?w=500", desc: "Ausführung: Rücken flach gegen eine echte Wand pressen. Nach unten rutschen, bis die Knie einen 90-Grad-Winkel bilden. Die Oberschenkel stehen parallel zum Boden." },
          { title: "2. Kniebeugen (Squats)", imgUrl: "https://images.unsplash.com/photo-1574680096145-d05b474e2155?w=500", desc: "Ausführung: Schulterbreiter Stand. Schiebe das Gesäß kontrolliert nach hinten-unten, als ob du dich auf einen unsichtbaren Stuhl setzen willst." },
          { title: "3. Unterarmstütz (Plank)", imgUrl: "https://images.unsplash.com/photo-1566241477600-ac026ad43874?w=500", desc: "Ausführung: Auf den Unterarmen abstützen. Der Körper muss vom Kopf bis zu den Fersen eine absolut gerade Linie bilden. Bauchdecke fest einziehen." }
        ];

        let completed = [];
        let activeTab = 'heute';
        let subTab = 'uebungen'; 
        let planPhase = 1;
        let planWeek = 0;

        const ALL_SESSIONS = [];
        ALL_WEEKS.forEach((w, wi) => {
          w.sessions.forEach((s, si) => {
            ALL_SESSIONS.push({ weekIdx: wi, sessionIdx: si, week: w.week, phase: w.phase, focus: w.focus, session: s, tip: w.tip });
          });
        });
        const TOTAL = ALL_SESSIONS.length;

        function isCompleted(idx) { return completed.includes(idx); }
        function toggleCompleted(idx) {
          completed.includes(idx) ? completed = completed.filter(i => i !== idx) : completed.push(idx);
          render();
        }
        function getCurrentIdx() {
          for (let i = 0; i < TOTAL; i++) { if (!completed.includes(i)) return i; }
          return TOTAL;
        }

        function renderSession(s, globalIdx, showCheck) {
          const sc = TYPE_COLOR[s.session.type] || { bg:"#f5f5f5", border:"#999", icon:"📋" };
          const done = isCompleted(globalIdx);

          let bodyHTML = s.session.content ? `<p style="font-size:13px; line-height:1.5; color:#333;">${s.session.content}</p>` : s.session.blocks.map(b => `
            <div style="margin-bottom:12px;">
              <div class="block-label">${b.label}</div>
              ${b.items.map(item => `<div class="block-item"><span class="block-dot" style="color:${sc.border}">·</span><span>${item}</span></div>`).join('')}
            </div>
          `).join('');

          return `
            <div class="session-card" style="${globalIdx === getCurrentIdx() ? 'border-color: #2196f3;' : ''}">
              <div class="session-head" style="background:${done ? '#f1f8e9' : sc.bg}; border-left-color:${done ? '#4caf50' : sc.border};">
                <span class="session-head-left">${done ? '✓' : sc.icon} ${s.session.day} – ${s.session.type}</span>
                <span class="session-head-right">${s.session.duration}</span>
              </div>
              <div class="session-body">${bodyHTML}</div>
              ${showCheck ? `
                <div style="padding: 0 14px 14px;">
                  <button class="check-btn ${done ? 'done-btn' : 'todo'}" onclick="toggleCompleted(${globalIdx})">
                    ${done ? '✓ Erledigt' : 'Als erledigt markieren'}
                  </button>
                </div>` : ''}
            </div>
          `;
        }

        function render() {
          const currentIdx = getCurrentIdx();
          const pct = Math.round((completed.length / TOTAL) * 100) || 0;
          let content = '';

          if (activeTab === 'heute') {
            if (currentIdx >= TOTAL) {
              content = `<div class="today-wrap"><div class="done-banner">🏆 <h2>Plan beendet!</h2></div></div>`;
            } else {
              const cur = ALL_SESSIONS[currentIdx];
              content = `
                <div class="today-wrap">
                  <div class="current-card">
                    <div class="current-label">Nächste Einheit</div>
                    <div class="current-week">Phase ${cur.phase} · Woche ${cur.week}</div>
                    <div class="current-title">${cur.focus}</div>
                  </div>
                  ${renderSession(cur, currentIdx, true)}
                  <div class="tip-box">
                    <div class="tip-label">💡 Tipp der Woche</div>
                    <div class="tip-text">${cur.tip}</div>
                  </div>
                </div>`;
            }
          } else if (activeTab === 'plan') {
            const phaseWeeks = ALL_WEEKS.filter(w => w.phase === planPhase);
            const weekData = phaseWeeks[planWeek] || phaseWeeks[0];
            const weekGlobalStart = ALL_WEEKS.indexOf(weekData) * 3;

            content = `
              <div class="plan-wrap">
                <div class="phase-tabs">
                  ${[1,2,3].map(p => `<button class="phase-tab ${p===planPhase?'active':''}" onclick="planPhase=${p};planWeek=0;render()">Phase ${p}</button>`).join('')}
                </div>
                <div class="week-btns">
                  ${phaseWeeks.map((w, i) => `<button class="week-btn ${planWeek===i?'active':''}" onclick="planWeek=${i};render()">Woche ${w.week}</button>`).join('')}
                </div>
                <div class="week-header">
                  <div class="week-header-sub">Woche ${weekData.week}</div>
                  <div class="week-header-title">${weekData.focus}</div>
                </div>
                ${weekData.sessions.map((s, si) => renderSession({session:s}, weekGlobalStart + si, true)).join('')}
              </div>`;
          } else {
            let subContent = '';
            if (subTab === 'uebungen') {
              subContent = EXERCISE_LIBRARY.map(ex => `
                <div class="tip-card">
                  <div class="tip-card-title">${ex.title}</div>
                  <div class="img-container">
                    <img src="${ex.imgUrl}" alt="${ex.title}">
                  </div>
                  <div class="exercise-desc">${ex.desc}</div>
                </div>
              `).join('');
            } else {
              subContent = VEGAN_NUTRITION_TIPS.map(t => `
                <div class="tip-card">
                  <div style="font-weight:700; color:#1a1a2e; margin-bottom:6px;">${t.icon} ${t.title}</div>
                  <div style="font-size:13px; color:#444; line-height:1.6;">${t.text}</div>
                </div>
              `).join('');
            }

            content = `
              <div class="tips-wrap">
                <div class="sub-tabs">
                  <button class="sub-tab ${subTab==='uebungen'?'active':''}" onclick="subTab='uebungen';render()">🏋️‍♂️ Mediathek</button>
                  <button class="sub-tab ${subTab==='ernaehrung'?'active':''}" onclick="subTab='ernaehrung';render()">🥑 Ernährung</button>
                </div>
                ${subContent}
              </div>`;
          }

          document.getElementById('root').innerHTML = `
            <div class="header">
              <div class="header-sub">12-Wochen Laufprogramm</div>
              <h1>Vom Sofa zur 30-Min-Runde</h1>
            </div>
            <div class="progress-bar-wrap">
              <div class="progress-label"><span>Fortschritt</span><span>${completed.length} / ${TOTAL} Einheiten (${pct}%)</span></div>
              <div class="progress-track"><div class="progress-fill" style="width:${pct}%"></div></div>
            </div>
            <div class="tabs">
              <button class="tab ${activeTab==='heute'?'active':''}" onclick="activeTab='heute';render()">Heute</button>
              <button class="tab ${activeTab==='plan'?'active':''}" onclick="activeTab='plan';render()">Alle Wochen</button>
              <button class="tab ${activeTab==='tipps'?'active':''}" onclick="activeTab='tipps';render()">Tipps & Übungen</button>
            </div>
            ${content}
          `;
        }

        render();
    </script>
</body>
